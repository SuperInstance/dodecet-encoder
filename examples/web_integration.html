📖 What is Dodecet Encoding?

A dodecet is a 12-bit unit (3 nibbles) optimized for geometric and calculus operations. It provides 4096 discrete states, perfect for encoding coordinates, vectors, and transformations.

💾 Memory Efficient

6 bytes per 3D point vs 24 bytes for f64 (75% savings)

🎯 Deterministic

No floating-point drift or rounding errors

⚡ Fast Operations

Integer math is faster than floating-point

🔒 Safe

Bounded range prevents overflow errors

💻 Integration Examples

1. Basic Usage (TypeScript/JavaScript)

// Import the library (when published to npm)
import { DodecetPoint3D } from '@superinstance/dodecet-encoder';

// Create a 3D point
const point = new DodecetPoint3D(1000, 2000, 3000);

// Access coordinates
console.log('X:', point.x);      // 1000
console.log('Y:', point.y);      // 2000
console.log('Z:', point.z);      // 3000

// Convert to hex string
console.log('Hex:', point.toHex());  // "3E87BB8"

// Calculate distance to another point
const other = new DodecetPoint3D(1500, 2500, 3500);
console.log('Distance:', point.distanceTo(other));  // 866.0

// Normalize to [0, 1] range
const [nx, ny, nz] = point.normalized();
console.log('Normalized:', nx, ny, nz);  // 0.244, 0.488, 0.732

2. Pythagorean Snapping

import { PythagoreanSnapper } from '@superinstance/dodecet-encoder';

// Create snapper with 12-bit precision
const snapper = new PythagoreanSnapper();

// Snap point to nearest Pythagorean triple
const point = new DodecetPoint3D(1000, 2000, 3000);
const snapped = snapper.snap(point);

console.log('Original:', point.toHex());     // "3E87BB8"
console.log('Snapped:', snapped.toHex());     // "3E87BB8"
console.log('Triple:', snapper.getTriple(snapped));  // "(3, 4, 5)"

4. Web Workers for Parallel Processing

// worker.ts
import { DodecetPoint3D, PythagoreanSnapper } from '@superinstance/dodecet-encoder';

self.onmessage = (e) => {
    const points = e.data;
    const snapper = new PythagoreanSnapper();

    const results = points.map(p => {
        const point = new DodecetPoint3D(p.x, p.y, p.z);
        return {
            original: point.toHex(),
            snapped: snapper.snap(point).toHex(),
            triple: snapper.getTriple(point)
        };
    });

    postMessage(results);
};

// main.ts
const worker = new Worker('worker.ts', { type: 'module' });

const points = [
    { x: 1000, y: 2000, z: 3000 },
    { x: 1500, y: 2500, z: 3500 },
    { x: 2000, y: 3000, z: 4000 }
];

worker.onmessage = (e) => {
    console.log('Results:', e.data);
};

worker.postMessage(points);

5. React Component Example

import React, { useState } from 'react';
import { DodecetPoint3D } from '@superinstance/dodecet-encoder';

function PointVisualizer() {
    const [point, setPoint] = useState(new DodecetPoint3D(1000, 2000, 3000));

    const updateX = (x) => {
        setPoint(new DodecetPoint3D(x, point.y, point.z));
    };

    return (
        <div>
            <h3>Point: {point.toHex()}</h3>
            <div>X: {point.x}</div>
            <div>Y: {point.y}</div>
            <div>Z: {point.z}</div>
            <div>Normalized: ({point.normalized().join(', ')})</div>
            <input
                type="range"
                min="0"
                max="4095"
                value={point.x}
                onChange={(e) => updateX(parseInt(e.target.value))}
            />
        </div>
    );
}

🚀 SuperInstance Integration

Constraint Theory

// Pythagorean snapping for constraint theory
import { PythagoreanSnapper, DodecetPoint3D } from '@superinstance/dodecet-encoder';

// Φ-Folding Operator: Continuous → Discrete
const snapper = new PythagoreanSnapper();
const continuousPoint = { x: 1.5, y: 2.7, z: 3.2 };

// Normalize and snap
const dodecetPoint = new DodecetPoint3D(
    Math.floor(continuousPoint.x * 4095),
    Math.floor(continuousPoint.y * 4095),
    Math.floor(continuousPoint.z * 4095)
);

const snapped = snapper.snap(dodecetPoint);
console.log('Snapped to triple:', snapper.getTriple(snapped));

Cellular Agents

// Use dodecet encoding for cellular agent state
import { DodecetArray } from '@superinstance/dodecet-encoder';

// Agent state encoded as dodecets
class ClawAgent {
    state: DodecetArray<4>;

    constructor() {
        // Position (3 dodecets) + Status (1 dodecet)
        this.state = new DodecetArray([0x000, 0x000, 0x000, 0x001]);
    }

    moveTo(x: number, y: number, z: number) {
        this.state = new DodecetArray([x, y, z, this.state[3]]);
    }

    getStatus(): string {
        const status = this.state[3];
        return status === 0x001 ? 'ACTIVE' : 'INACTIVE';
    }
}
